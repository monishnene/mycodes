int max_of_four(int a, int b, int c, int d)
{
    int n=a;
    n=n>b?n:b;
    n=n>c?n:c;
    n=n>d?n:d;
    return n;
}

void update(int *a,int *b)
{
    int A=*a;
    int B=*b;
    *(a)=A+B;
    *(b)=A-B;
    if(*(a)<0)
    {
        *(a)*=-1;
    }
    if(*(b)<0)
    {
        *(b)*=-1;
    }
}

int double_array() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    long j=0,i=0,rows=0,queries=0,columns=0,row=0,column=0;
    cin>>rows;
    cin>>queries;
    long* arrays[rows];
    for(i=0;i<rows;i++)
    {
        cin>>columns;
        arrays[i]=(long*)malloc(columns*sizeof(long*));
        for(j=0;j<columns;j++)
        {
            cin>>*(arrays[i]+j);
        }
    }
    for(i=0;i<queries;i++)
    {
        cin>>row;
        cin>>column;
        printf("%ld\n",*(arrays[row]+column));
    }
    for(i=0;i<rows;i++)
    {
        free(arrays[i]);
    }
    return 0;
}
